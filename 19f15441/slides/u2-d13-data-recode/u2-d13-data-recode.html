.midi[]

Save data

rel_inc_long <- rel_inc %>% 
  rename(
    religion = `Religious tradition`,
    n = `Sample Size`
  ) %>%
  pivot_longer(
    cols = -c(religion, n), 
    names_to = "income", 
    values_to = "proportion"
  ) %>%
  mutate(frequency = round(proportion * n))

Barplot

ggplot(rel_inc_long, aes(y = religion, x = frequency)) +
  geom_col()
